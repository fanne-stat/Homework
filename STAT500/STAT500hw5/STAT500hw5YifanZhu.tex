		Y_{31} - \bar{Y}_{..}\\

	\end{bmatrix}
	\\
	&= \sum_{i = 1}^3 \sum_{j = 1}^3 Y_{ij}(Y_{ij} - \bar{Y}_{..})\\

	\end{align*}
